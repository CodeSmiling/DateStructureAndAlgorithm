## 归并排序算法

### 稳定性：稳定排序算法

### 平均时间复杂度：O(NlogN)
---
```java
/**
 * 归并排序--采用分治的思想
 * 1.先分，将整个数组分为一个个单独的数组
 * 2.将独立数组进行合并
 */
public class MergeSort {
    public void sort(int []arr){
        int length=arr.length;
        //建立一个临时数组
        int[] temp=new int[length];
        sort(arr,0,length-1,temp);
    }
    public void sort(int[] arr,int start,int end,int[] temp){
        if(start<end){
            int mid=(start+end)/2;
            sort(arr,start,mid,temp);//左边数组进行分割
            sort(arr,mid+1,end,temp);//右边数组进行分割
            merge(arr,start,mid,end,temp);
        }

    }
    //将分割的数组进行合并
    public void merge(int[] arr, int start, int mid, int end, int[] temp) {
        //左边序列的指针
        int l=start;
        //右边序列的指针
        int r=mid+1;
        //临时数组的指针
        int j=0;
        //比较两个分开数组的大小
        while(l<=mid&&r<=end){
            if(arr[l]<=arr[j])
                temp[j++]=arr[l++];
            else {
                temp[j++]=arr[r++];
            }
        }
        //将剩余部分填入临时数组
        while (l<=mid){
            temp[j++]=arr[l++];
        }
        while (r<=end){
            temp[j++]=arr[r++];
        }
        //将temp中的元素全部拷贝进原数组
        while (start<=end){
            arr[start++]=temp[j++];
        }
    }
}
```
