Q:题目链接https://leetcode-cn.com/problems/fu-za-lian-biao-de-fu-zhi-lcof/

A:整个过程分为三步
  1.在源节点之后复制生成一个源节点
  
  2.将源节点的random值依旧复制到新生成的节点后
  
  3.重新生成一个复制节点的链表
  ```java
  class Node{
    int val;
    Node next;
    Node random;
    public Node(int x){
        int val=x;
        Node head=null;
        Node random=null;
    }
}
class Solution{
    public Node copyRandomList(Node head){
        if(head==null) return null;
        //复制节点
        copy(head);
        //寻找random节点
        randomDirect(head);
        //返回复制链表;
        return reList(head);
    }

    public void copy(Node head){
        while(head!=null){
             Node cloneNode=new Node(head.val);
             Node nextNode=head.next;
             head.next=cloneNode;
             cloneNode.next=nextNode;
             //递推公式
             head=cloneNode.next;
        }
    }
    public void randomDirect(Node head){
        while(head!=null){
            Node cloneNode=head.next;
            if(head.random!=null){
                Node direct=head.random;
                cloneNode.random=direct.next;
            }
            head=cloneNode.next;
        }
    }
    public Node reList(Node head){
        Node cloneNode=head.next;
        Node cloneHead=cloneNode;
        head.next=cloneNode.next;
        head=head.next;
        while(head!=null){
            /**
             *链表递推过程
             * 1.指定当前节点的下一个节点
             * 2.然后将当前节点位移到下个节点
             */
            cloneNode.next=head.next;
            head.next=head.next.next;
            head=head.next;
            cloneNode=cloneNode.next;
        }
        return cloneHead;
    }
}
  ```
